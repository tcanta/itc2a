Programmation Dynamique#

Il sera judicieux de comparer les temps d’exécution afin de comparer les performances des différents algorithmes.

import time

Factorielle#

def fact(n):
    if n == 0: # cas de base
        return 1
    return n*fact(n - 1) # appel récursif

start = time.time()
print(f"Value: {fact(42)}")
end = time.time()
duration = (end - start)*1000
print(f"Duration: {duration} ms")

Version dynamique

def fibo(n):
    L = [1, 1] # L[i] va contenir le ième terme de Fibonacci
    for i in range(n - 1):
        L.append(L[-1] + L[-2]) # récurrence
    return L[n]

def fibo(n):
    f0, f1 = 1, 1 # f0, f1 sont les deux derniers termes
    for i in range(n - 1):
        f0, f1 = f1, f0 + f1
    return f1

start = time.time()
print(f"Value: {fibo(32)}")
end = time.time()
duration = (end - start)*1000
print(f"Duration: {duration} ms")

Value: 3524578
Duration: 0.06937980651855469 ms

Coefficients binomiaux#

Version naive récursive

def binom(n, k):
    if k == 0: return 1
    if n == 0: return 0
    return binom(n - 1, k - 1) + binom(n - 1, k)

start = time.time()
print(f"Value: {binom(30,8)}")
end = time.time()
duration = (end - start)*1000

Version dynamique

def binom(n, k):
    M = [[0]*(k + 1) for _ in range(n + 1)]
    for i in range(0, n + 1):
        M[i][0] = 1
    for i in range(1, n + 1):
        for j in range(1, k + 1):
            M[i][j] = M[i - 1][j - 1] + M[i - 1][j]
    return M[n][k]

start = time.time()
print(f"Value: {binom(30,8)}")
end = time.time()
duration = (end - start)*1000
print(f"Duration: {duration} ms")

Knapsack#

def knapsack(c, w, v):
    """
    Renvoie la valeur maximum que l'on peut mettre
    dans un sac à dos de capacité c.
    Le ième objet a pour poids w[i] et valeur v[i].
    """
    n = len(w) # nombre d'objets
    dp = [[0]*(n + 1) for i in range(c + 1)]
    # dp[i][j] = valeur max dans un sac de capacité i
    # où j est le nombre d'objets autorisés
    for i in range(1, c + 1):
        for j in range(1, n + 1):
            if w[j - 1] <= i:
                x = v[j - 1] + dp[i - w[j - 1]][j - 1]
                dp[i][j] = max(dp[i][j - 1], x)
            else:
                dp[i][j] = dp[i][j - 1]
    return dp[c][n]

knapsack(7,[1,2,2,4,5],[1,4,5,2,4])

def knapsack2(c, w, v):
    n = len(w)
    dp = [0]*(c + 1)
    for j in range(n):
        dp_ = dp[:] # copie de dp
        for i in range(c + 1):
            if w[j] <= i:
                dp[i] = max(dp_[i], v[j] + dp_[i - w[j]])
    return dp[-1]

def fibo(n):
    d = {} # d[k] contiendra le kème terme de la suite
    def aux(k):
        if k == 0 or k == 1:
            return 1
        if k not in d:
            d[k] = aux(k - 1) + aux(k - 2)
        return d[k]
    return aux(n)

Coefficients Binomiaux#

def binom(n, k):
    d = {}
    def aux(i, j):
        if j == 0: return 1
        if i == 0: return 0
        if (i, j) not in d:
            d[(i, j)] = aux(i - 1, j - 1) + aux(i - 1, j)
        return d[(i, j)]
    return aux(n, k)

start = time.time()

end = time.time()
duration = (end - start)*1000
print(f"Duration: {duration} ms")

Knapsack#

def knapsack_memo(c, w, v):
    dp = {}
    def aux(i, j):
        if i == 0 or j == 0:
            return 0
        if (i, j) not in dp:
            dp[(i, j)] = aux(i, j - 1)
            if w[j - 1] <= i:
                x = v[j - 1] + aux(i - w[j - 1], j - 1)
                dp[(i, j)] = max(dp[(i, j)], x)
        return dp[(i, j)]
    return aux(c, len(w))
